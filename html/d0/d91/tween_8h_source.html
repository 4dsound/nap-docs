229  void Tween<T>::setDuration(float duration)
230  {
231  assert(duration >= 0.0f); // invalid duration
232 
233  // when duration is bigger then 0, scale time accordingly to ensure smooth transition
234  if (duration > 0.0f)
235  {
236  float current_progress = mTime / mDuration;
237  mDuration = duration;
238  mTime = mDuration * current_progress;
239  }
240  else
241  {
242  // when duration is 0, it doesn't matter since we will hit complete in next update
243  mDuration = duration;
244  mTime = 0.0f;
245  }
246  }

248  template<typename T>
249  void Tween<T>::setMode(ETweenMode mode)
250  {
251  mMode = mode;
252 
253  switch (mode)
254  {
255  default:
256  nap::Logger::warn("Unknown tween mode, choosing NORMAL mode");
257  case NORMAL:
258  {
259  mUpdateFunc = [this](double deltaTime)
260  {
261  if (!mComplete)
262  {
263  mTime += deltaTime;
264 
265  if (mTime >= mDuration)
266  {
267  mTime = mDuration;
268  mComplete = true;
269 
270  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
271 
272  UpdateSignal.trigger(mCurrentValue);
273  CompleteSignal.trigger(mCurrentValue);
274  }
275  else
276  {
277  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
278  UpdateSignal.trigger(mCurrentValue);
279  }
280  }
281  };
282  }
283  break;
284  case PING_PONG:
285  {
286  float direction = 1.0f;
287  mUpdateFunc = [this, direction](double deltaTime) mutable
288  {
289  mTime += deltaTime * direction;
290 
291  if (mTime >= mDuration)
292  {
293  mTime = mDuration - (mTime - mDuration);
294  direction = -1.0f;
295 
296  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
297  UpdateSignal.trigger(mCurrentValue);
298  }
299  else if (mTime <= 0.0f)
300  {
301  mTime = -mTime;
302  direction = 1.0f;
303 
304  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
305  UpdateSignal.trigger(mCurrentValue);
306  }
307  else
308  {
309  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
310  UpdateSignal.trigger(mCurrentValue);
311  }
312  };
313  }
314  break;
315  case LOOP:
316  {
317  mUpdateFunc = [this](double deltaTime)
318  {
319  if (!mComplete)
320  {
321  mTime += deltaTime;
322 
323  if (mTime >= mDuration)
324  {
325  mTime = mDuration - mTime;
326 
327  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
328  UpdateSignal.trigger(mCurrentValue);
329  }
330  else
331  {
332  mCurrentValue = mEase->evaluate(mStart, mEnd, mTime / mDuration);
333  UpdateSignal.trigger(mCurrentValue);
334  }
335  }
336  };
337  }
338  break;
339  case REVERSE:
340  {
341  mUpdateFunc = [this](double deltaTime)
342  {
343  if (!mComplete)
344  {
345  mTime += deltaTime;
346 
347  if (mTime >= mDuration)
348  {
349  mComplete = true;
350  mTime = mDuration;
351 
352  mCurrentValue = mEase->evaluate(mStart, mEnd, 1.0f - (mTime / mDuration));
353 
354  UpdateSignal.trigger(mCurrentValue);
355  CompleteSignal.trigger(mCurrentValue);
356  }
357  else
358  {
359  mCurrentValue = mEase->evaluate(mStart, mEnd, 1.0f - (mTime / mDuration));
360  UpdateSignal.trigger(mCurrentValue);
361  }
362  }
363  };
364  }
365  break;
366  }
367  }
