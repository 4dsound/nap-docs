260  template<typename T>
261  void SequenceEventTrackView::handleEditEventSegmentPopup()
262  {
263  if (mState.mAction->isAction<sequenceguiactions::OpenEditEventSegmentPopup<T>>())
264  {
265  // invoke insert sequence popup
266  ImGui::OpenPopup("Edit Event");
267 
268  auto* action = mState.mAction->getDerived<sequenceguiactions::OpenEditEventSegmentPopup<T>>();
269  ImGui::SetNextWindowPos(action->mWindowPos);
270 
271  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::EditingEventSegment<T>>(action->mTrackID, action->mSegmentID, action->mWindowPos, action->mValue, action->mStartTime);
272  }
273 
274  // handle insert segment popup
275  if (mState.mAction->isAction<sequenceguiactions::EditingEventSegment<T>>())
276  {
277  auto* action = mState.mAction->getDerived<sequenceguiactions::EditingEventSegment<T>>();
278 
279  if (ImGui::BeginPopup("Edit Event"))
280  {
281  // draw the registered popup content for this event
282  auto it = mSegmentViews.find(RTTI_OF(SequenceTrackSegmentEvent<T>));
283  assert(it!= mSegmentViews.end()); // type not found
284  it->second->handleEditPopupContent(*action);
285 
286  // time
287  std::vector<int> time_array = convertTimeToMMSSMSArray(action->mStartTime);
288 
289  bool edit_time = false;
290 
291  ImGui::Separator();
292  ImGui::PushItemWidth(100.0f * mState.mScale);
293 
294  edit_time = ImGui::InputInt3("Time (mm:ss:ms)", &time_array[0]);
295  time_array[0] = math::clamp<int>(time_array[0], 0, 99999);
296  time_array[1] = math::clamp<int>(time_array[1], 0, 59);
297  time_array[2] = math::clamp<int>(time_array[2], 0, 99);
298 
299  ImGui::PopItemWidth();
300 
301  ImGui::Separator();
302 
303  if(edit_time)
304  {
305  auto& event_controller = getEditor().getController<SequenceControllerEvent>();
306  double new_time = convertMMSSMSArrayToTime(time_array);
307  double time = event_controller.segmentEventStartTimeChange(action->mTrackID, action->mSegmentID, new_time);
308  updateSegmentInClipboard(action->mTrackID, action->mSegmentID);
309  action->mStartTime = time;
310  mState.mDirty = true;
311  }
312  if (ImGui::ImageButton(mService.getGui().getIcon(icon::ok)))
313  {
314  auto& event_controller = getEditor().getController<SequenceControllerEvent>();
315  event_controller.editEventSegment<T>(action->mTrackID, action->mSegmentID, action->mValue);
316  updateSegmentInClipboard(action->mTrackID, action->mSegmentID);
317  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
318  ImGui::CloseCurrentPopup();
319  }
320  ImGui::SameLine();
321  if (ImGui::ImageButton(mService.getGui().getIcon(icon::cancel)))
322  {
323  ImGui::CloseCurrentPopup();
324  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
325  }
326 
327  action->mWindowPos = ImGui::GetWindowPos();
328 
329  ImGui::EndPopup();
330  }
331  else
332  {
333  // click outside popup so cancel action
334  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
335  }
336  }
