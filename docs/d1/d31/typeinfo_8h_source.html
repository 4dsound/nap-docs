197  inline bool hasFlag(const rtti::Property& property, EPropertyMetaData flags)
198  {
199  const rtti::Variant& meta_data = property.get_metadata("flags");
200  if (!meta_data.is_valid())
201  return false;
202 
203  uint8_t current_flags = meta_data.convert<uint8_t>();

224  inline rttr::method findMethodRecursive(const rtti::TypeInfo& type, const std::string& methodName)
225  {
