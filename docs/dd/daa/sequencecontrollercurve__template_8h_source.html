270  template<typename T>
271  void SequenceControllerCurve::insertCurvePoint(SequenceTrackSegment& segment, float pos, int curveIndex)
272  {
273  assert(segment.get_type().is_derived_from<SequenceTrackSegmentCurve<T>>()); // type mismatch
274 
275  auto& curve_segment = static_cast<SequenceTrackSegmentCurve<T>&>(segment);
276  assert(curveIndex<curve_segment.mCurves.size()); // invalid curve index
277 
278  // iterate trough points of curve
279  for (int i = 0; i<curve_segment.mCurves[curveIndex]->mPoints.size()-1; i++)
280  {
281  // find the point the new point needs to get inserted after
282  if (curve_segment.mCurves[curveIndex]->mPoints[i].mPos.mTime<=pos && curve_segment.mCurves[curveIndex]->mPoints[i+1].mPos.mTime>pos)
283  {
284  // create point
285  math::FCurvePoint<float, float> p;
286  p.mPos.mTime = pos;
287  p.mPos.mValue = curve_segment.mCurves[curveIndex]->evaluate(pos);
288  p.mInTan.mTime = -0.1f;
289  p.mOutTan.mTime = 0.1f;
290  p.mInTan.mValue = 0.0f;
291  p.mOutTan.mValue = 0.0f;
292  p.mTangentsAligned = true;
293  p.mInterp = curve_segment.mCurveTypes[curveIndex];
294 
295  // insert point
296  curve_segment.mCurves[curveIndex]->mPoints.insert(curve_segment.mCurves[curveIndex]->mPoints.begin()+i+1, p);
297  curve_segment.mCurves[curveIndex]->invalidate();
298  break;
299  }
300  }
301  }

361  template<typename T>
362  bool SequenceControllerCurve::changeTanPoint(SequenceTrackSegment& segment,
363  const std::string& trackID, const int pointIndex, const int curveIndex,
364  sequencecurveenums::ETanPointTypes tanType, float time, float value)
365  {
366  assert(segment.get_type().is_derived_from<SequenceTrackSegmentCurve<T>>()); // type mismatch
367  auto& curve_segment = static_cast<SequenceTrackSegmentCurve<T>&>(segment);
368  assert(curveIndex<curve_segment.mCurves.size()); // invalid curve index
369  assert(pointIndex<curve_segment.mCurves[curveIndex]->mPoints.size()); // invalid point index
370 
371  // if tangents are flipped, set this to true
372  bool flipped = false;
373 
374  // get reference to curve point
375  auto& curve_point = curve_segment.mCurves[curveIndex]->mPoints[pointIndex];
376  switch (tanType)
377  {
378  case sequencecurveenums::ETanPointTypes::IN:
379  {
380  // if we have aligned tangents, check if IN and OUT can be flipped
381  if (curve_point.mTangentsAligned && time>0.0f)
382  {
383  time = -time;
384  value = -value;
385  flipped = true;
386  }
387 
388  curve_point.mInTan.mTime = time;
389  curve_point.mInTan.mValue = value;
390 
391  if (curve_point.mTangentsAligned)
392  {
393  curve_point.mOutTan.mTime = -curve_point.mInTan.mTime;
394  curve_point.mOutTan.mValue = -curve_point.mInTan.mValue;
395  }
396  }
397  break;
398  case sequencecurveenums::ETanPointTypes::OUT:
399  {
400  if (curve_point.mTangentsAligned && time<0.0f)
401  {
402  time = -time;
403  value = -value;
404  flipped = true;
405  }
406 
407  curve_point.mOutTan.mTime = time;
408  curve_point.mOutTan.mValue = value;
409 
410  if (curve_point.mTangentsAligned)
411  {
412  curve_point.mInTan.mTime = -curve_point.mOutTan.mTime;
413  curve_point.mInTan.mValue = -curve_point.mOutTan.mValue;
414  }
415  }
416  break;
417  }
418 
419  // is this the last control point ?
420  // then also change the first control point of the next segment accordinly
421  if (pointIndex==curve_segment.mCurves[curveIndex]->mPoints.size()-1)
422  {
423  SequenceTrack* track = findTrack(trackID);
424  assert(track!=nullptr); // track not found
425 
426  for (int i = 0; i<track->mSegments.size(); i++)
427  {
428  if (track->mSegments[i].get()==&segment && i+1<track->mSegments.size())
429  {
430  auto& next_segment_curve_point = static_cast<SequenceTrackSegmentCurve<T>&>(*track->mSegments[i+1]).mCurves[curveIndex]->mPoints[0];
431 
432  next_segment_curve_point.mInTan.mTime = curve_point.mInTan.mTime;
433  next_segment_curve_point.mInTan.mValue = curve_point.mInTan.mValue;
434  next_segment_curve_point.mOutTan.mTime = curve_point.mOutTan.mTime;
435  next_segment_curve_point.mOutTan.mValue = curve_point.mOutTan.mValue;
436  }
437  }
438  }
439 
440  curve_segment.mCurves[curveIndex]->invalidate();
441 
442  return flipped;
443  }
