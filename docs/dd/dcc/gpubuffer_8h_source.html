498  template<typename T>
499  bool TypedGPUBufferNumeric<T>::init(utility::ErrorState& errorState)
500  {
501  // Ensure storage bit is set if requested
502  VkBufferUsageFlags req_usage = mStorage ? VK_BUFFER_USAGE_STORAGE_BUFFER_BIT : 0;
503  this->ensureUsage(req_usage);
504 
505  // Initialize Base
506  if (!GPUBufferNumeric::init(errorState))
507  return false;
508 
509  if (!errorState.check(mMemoryUsage != EMemoryUsage::DynamicWrite || mCount >= 0,
510  "Cannot allocate a non-DynamicWrite buffer with zero elements."))
511  return false;
512 
513  // Calculate buffer size
514  uint32 buffer_size = mCount * sizeof(T);
515 
516  // Allocate buffer memory
517  if (!allocateInternal(buffer_size, errorState))
518  return false;
519 
520  // Upload data when a buffer fill policy is available
521  if (mFillPolicy != nullptr)
522  {
523  if (mMemoryUsage != EMemoryUsage::DynamicRead)
524  {
525  // Create a staging buffer to upload
526  auto staging_buffer = std::make_unique<T[]>(mCount);
527  mFillPolicy->fill(mCount, staging_buffer.get());
528 
529  // Prepare staging buffer upload
530  if (!setDataInternal(staging_buffer.get(), buffer_size, buffer_size, errorState))
531  return false;
532  }
533  else
534  {
535  // Warn user that buffers cannot be filled when their usage is set to DynamicRead
536  nap::Logger::warn(utility::stringFormat("%s: The configured fill policy was ignored as the buffer usage is DynamicRead", mID.c_str()).c_str());
537  }
538  }
539 
540  // Optionally clear - does not count as an upload
541  else if (mClear)
542  {
543  GPUBuffer::requestClear();
544  }
545 
546  mInitialized = true;
547  return true;
548  }
