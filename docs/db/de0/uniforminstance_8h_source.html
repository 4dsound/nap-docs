409  template<typename T>
410  T* nap::UniformStructInstance::getOrCreateUniform(const std::string& name)
411  {
412  // First try to find it, if found cast and return
413  UniformInstance* instance = findUniform(name);
414  if (instance != nullptr)
415  {
416  assert(instance->get_type().is_derived_from<T>());
417  return rtti_cast<T>(instance);
418  }
419 
420  // Otherwise fetch the declaration and use it to create the new instance
421  const ShaderVariableDeclaration* declaration = mDeclaration.findMember(name);
422  if (declaration == nullptr)
423  return nullptr;
424 
425  std::unique_ptr<UniformInstance> new_instance = createUniformFromDeclaration(*declaration, mUniformCreatedCallback);
426  T* result = rtti_cast<T>(new_instance.get());
427  assert(result != nullptr);
428  mUniforms.emplace_back(std::move(new_instance));
429 
430  // Notify listeners
431  if (mUniformCreatedCallback)
432  mUniformCreatedCallback();
433  return result;
434  }
