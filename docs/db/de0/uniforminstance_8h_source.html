399  template<typename T>
400  T* nap::UniformStructInstance::getOrCreateUniform(const std::string& name)
401  {
402  // First try to find it, if found cast and return
403  UniformInstance* instance = findUniform(name);
404  if (instance != nullptr)
405  {
406  assert(instance->get_type().is_derived_from<T>());
407  return rtti_cast<T>(instance);
408  }
409 
410  // Otherwise fetch the declaration and use it to create the new instance
411  const UniformDeclaration* declaration = mDeclaration.findMember(name);
412  if (declaration == nullptr)
413  return nullptr;
414 
415  std::unique_ptr<UniformInstance> new_instance = createUniformFromDeclaration(*declaration, mUniformCreatedCallback);
416  T* result = rtti_cast<T>(new_instance.get());
417  assert(result != nullptr);
418  mUniforms.emplace_back(std::move(new_instance));
419 
420  // Notify listeners
421  if (mUniformCreatedCallback)
422  mUniformCreatedCallback();
423  return result;
424  }
