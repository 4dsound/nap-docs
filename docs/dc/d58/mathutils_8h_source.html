310  template<typename T>
311  float fit(T value, T min, T max, T outMin, T outMax)
312  {
313  T v = glm::clamp<T>(value, min, max);
314  T m = max - min;
315  m = (m == 0.0f) ? std::numeric_limits<T>::epsilon() : m;
316  return (v - min) / (m) * (outMax - outMin) + outMin;
317  }

380  T smoothStep(T value, T edge0, T edge1)
381  {
382  T x = math::clamp<T>((value - edge0) / (edge1 - edge0), 0, 1);
