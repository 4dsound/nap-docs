16  template<typename T>
17  void SequenceCurveTrackView::handleCurvePointActionPopup()
18  {
19  if(mState.mAction->isAction<sequenceguiactions::OpenCurvePointActionPopup<T>>())
20  {
21  auto *action = mState.mAction->getDerived<sequenceguiactions::OpenCurvePointActionPopup<T>>();
22  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::CurvePointActionPopup<T>>(
23  action->mTrackID,
24  action->mSegmentID,
25  action->mControlPointIndex,
26  action->mCurveIndex,
27  action->mValue,
28  action->mTime,
29  action->mMinimum,
30  action->mMaximum);
31  ImGui::OpenPopup("Curve Point Actions");
32  }
33 
34  if(mState.mAction->isAction<sequenceguiactions::CurvePointActionPopup<T>>())
35  {
36  if(ImGui::BeginPopup("Curve Point Actions"))
37  {
38  auto *action = mState.mAction->getDerived<sequenceguiactions::CurvePointActionPopup<T>>();
39  int curveIndex = action->mCurveIndex;
40 
41  float value = action->mValue * (action->mMaximum[curveIndex] - action->mMinimum[curveIndex]) +
42  action->mMinimum[curveIndex];
43  if(ImGui::InputFloat("value", &value))
44  {
45  auto &curve_controller = getEditor().getController<SequenceControllerCurve>();
46  curve_controller.changeCurvePoint(
47  action->mTrackID,
48  action->mSegmentID,
49  action->mControlPointIndex,
50  action->mCurveIndex,
51  action->mTime,
52  value);
53  updateSegmentInClipboard(action->mTrackID, action->mSegmentID);
54 
55  mState.mDirty = true;
56  }
57 
63  // obtain segment
64  auto &curve_controller = getEditor().getController<SequenceControllerCurve>();
65  const auto *segment = curve_controller.getSegment(action->mTrackID, action->mSegmentID);
66  assert(segment != nullptr);
67 
68  double time = action->mTime * segment->mDuration + segment->mStartTime;
69  double min_time = segment->mStartTime;
70  double max_time = segment->mStartTime + segment->mDuration;
71 
72  std::vector<int> time_array = convertTimeToMMSSMSArray(time);
73 
74  bool edit_time = false;
75 
76  ImGui::Separator();
77  ImGui::PushItemWidth(100.0f * mState.mScale);
78 
79  edit_time = ImGui::InputInt3("Time (mm:ss:ms)", &time_array[0]);
80  time_array[0] = math::clamp<int>(time_array[0], 0, 99999);
81  time_array[1] = math::clamp<int>(time_array[1], 0, 59);
82  time_array[2] = math::clamp<int>(time_array[2], 0, 99);
83 
84  if(edit_time)
85  {
86  double new_time = convertMMSSMSArrayToTime(time_array);
87  new_time = math::clamp(new_time, min_time, max_time);
88 
89  float perc = (new_time - segment->mStartTime) / segment->mDuration;
90  action->mTime = perc;
91  curve_controller.changeCurvePoint(
92  action->mTrackID,
93  action->mSegmentID,
94  action->mControlPointIndex,
95  action->mCurveIndex,
96  perc,
97  action->mValue);
98  updateSegmentInClipboard(action->mTrackID, action->mSegmentID);
99  mState.mDirty = true;
100  }
101 
102  if(ImGui::ImageButton(mService.getGui().getIcon(nap::icon::del)))
103  {
104  auto &curve_controller = getEditor().getController<SequenceControllerCurve>();
105  curve_controller.deleteCurvePoint(
106  action->mTrackID,
107  action->mSegmentID,
108  action->mControlPointIndex,
109  action->mCurveIndex);
110  updateSegmentInClipboard(action->mTrackID, action->mSegmentID);
111  mCurveCache.clear();
112 
113  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
114  mState.mDirty = true;
115 
116  ImGui::CloseCurrentPopup();
117  }
118 
119  ImGui::SameLine();
120 
121  if(ImGui::ImageButton(mService.getGui().getIcon(nap::icon::ok)))
122  {
123  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
124  ImGui::CloseCurrentPopup();
125  }
126 
127  ImGui::EndPopup();
128  } else
129  {
130  // click outside popup so cancel action
131  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
132  }
133  }
134  }

198  template<typename T>
199  void SequenceCurveTrackView::drawCurves(
200  const SequenceTrack& track,
201  const SequenceTrackSegment& segmentBase,
202  const ImVec2& trackTopLeft,
203  const float previousSegmentX,
204  const float segmentWidth,
205  const float segmentX,
206  ImDrawList* drawList)
207  {
208  const auto &segment = static_cast<const SequenceTrackSegmentCurve<T> &>(segmentBase);
209  const float points_per_pixel = 0.5f * mState.mScale;
210  const float track_height = track.mTrackHeight * mState.mScale;
211  bool needs_drawing = ImGui::IsRectVisible({trackTopLeft.x + previousSegmentX, trackTopLeft.y}, {
212  trackTopLeft.x + previousSegmentX + segmentWidth, trackTopLeft.y + track_height});
213 
214  if(needs_drawing)
215  {
216  // if no cache present, create new curve
217  if(mCurveCache.find(segment.mID) == mCurveCache.end())
218  {
219  const long point_num = (int) (points_per_pixel * segmentWidth);
220  std::vector<std::vector<ImVec2>> curves;
221  for(int v = 0; v < segment.mCurves.size(); v++)
222  {
223  std::vector<ImVec2> curve;
224  if(point_num > 0)
225  {
226  auto start_x = math::max<float>(trackTopLeft.x, 0);
227  float end_x = start_x + mState.mWindowSize.x + mState.mWindowPos.x;
228 
229  // start drawing at window pos
230  {
231  long i = 0;
232  for(; i <= point_num; i++)
233  {
234  float p = (float) i / (float) point_num;
235  float x = trackTopLeft.x + previousSegmentX + segmentWidth * p;
236  if(x > start_x)
237  {
238  float value = 1.0f - segment.mCurves[v]->evaluate(p);
239  value = math::clamp(value, 0.0f, 1.0f);
240  ImVec2 point =
241  {
242  x,
243  trackTopLeft.y + value * track_height
244  };
245  curve.emplace_back(point);
246  }
247 
248  if(x > end_x)
249  {
250  break; // no longer visible on right side, continuation of this loop is not necessary
251  }
252  }
253  }
254  }
255 
256  curves.emplace_back(std::move(curve));
257  }
258  mCurveCache.emplace(segment.mID, std::move(curves));
259  }
260  }
261 
262  int selected_curve = -1;
263  if(mState.mIsWindowFocused)
264  {
265  // determine if mouse is hovering curve
266  if((mState.mAction->isAction<sequenceguiactions::None>() ||
267  mState.mAction->isAction<sequenceguiactions::HoveringCurve>())
268  && ImGui::IsMouseHoveringRect(
269  {trackTopLeft.x + segmentX - segmentWidth, trackTopLeft.y}, // top left
270  {trackTopLeft.x + segmentX, trackTopLeft.y + track_height})) // bottom right
271  {
272  // translate mouse position to position in curve
273  ImVec2 mouse_pos = ImGui::GetMousePos();
274  float x_in_segment =
275  ((mouse_pos.x - (trackTopLeft.x + segmentX - segmentWidth)) / mState.mStepSize) / segment.mDuration;
276  float y_in_segment = 1.0f - ((mouse_pos.y - trackTopLeft.y) / track_height);
277 
278  for(int i = 0; i < segment.mCurves.size(); i++)
279  {
280  // evaluate curve at x position
281  float y_in_curve = segment.mCurves[i]->evaluate(x_in_segment);
282 
283  // insert curve point on click
284  const float maxDist = 0.1f;
285  if(abs(y_in_curve - y_in_segment) < maxDist)
286  {
287  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::HoveringCurve>(
288  track.mID,
289  segment.mID,
290  i);
291 
292  if(ImGui::IsMouseClicked(1))
293  {
294  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::OpenInsertCurvePointPopup>(
295  track.mID,
296  segment.mID,
297  i, x_in_segment);
298  }
299  selected_curve = i;
300  }
301  }
302 
303  if(selected_curve == -1)
304  {
305  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
306  } else
307  {
308  showValue<T>(
309  track,
310  segment, x_in_segment,
311  mState.mMouseCursorTime, selected_curve);
312  }
313  } else
314  {
315  if(mState.mAction->isAction<sequenceguiactions::HoveringCurve>())
316  {
317  auto *action = mState.mAction->getDerived<sequenceguiactions::HoveringCurve>();
318 
319  if(action->mSegmentID == segment.mID)
320  {
321  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
322  }
323  }
324  }
325  }
326 
327  if(needs_drawing)
328  {
329  for(int i = 0; i < segment.mCurves.size(); i++)
330  {
331  if(mCurveCache[segment.mID][i].size() > 0)
332  {
333  float thickness = selected_curve == i ? 3.0f : 1.0f;
334  thickness *= mState.mScale;
335  drawList->AddPolyline
336  (
337  &*mCurveCache[segment.mID][i].begin(), // points array
338  mCurveCache[segment.mID][i].size(), // size of points array
339  mService.getColors().mCurveColors[i], // color
340  false, // closed
341  thickness // thickness
342  );
343  }
344  }
345  }
346  }

414  }
415  }
416  }
417 
418  // draw curve(s)
419  drawCurves<T>(
420  track,
421  segment,

424  segmentWidth,
425  segmentX,

427 
428

430  drawControlPoints<T>(
431  track,
432  segment,

434  segmentX,
435  segmentWidth,
436  drawList);
437 
438  // if this is the first segment of the track
439  // also draw a handler for the start value

441  {
442  // draw segment value handler
443  drawSegmentValue<T>(
444  track,
445  segment,
446  trackTopLeft,
447  segmentX,
448  segmentWidth,
449  sequencecurveenums::ESegmentValueTypes::BEGIN,
450  drawList);
451  }
452 
453  // draw segment value handler
454  drawSegmentValue<T>(
455  track,
456  segment,
457  trackTopLeft,
458  segmentX,
459  segmentWidth,
460  sequencecurveenums::ESegmentValueTypes::END,
461  drawList);
462  }

503  template<typename T>
504  void SequenceCurveTrackView::drawSegmentValue(
505  const SequenceTrack& track,
506  const SequenceTrackSegment& segmentBase,
507  const ImVec2& trackTopLeft,
508  const float segmentX,
509  const float segmentWidth,
510  const sequencecurveenums::ESegmentValueTypes segmentType,
511  ImDrawList* drawList)
512  {
513  static std::unordered_map<rttr::type, std::function<float(const SequenceTrackSegment &, int, sequencecurveenums::ESegmentValueTypes)>> get_value_map
514  {
515  {RTTI_OF(float), [](const SequenceTrackSegment &segment, int curveIndex, sequencecurveenums::ESegmentValueTypes segmentType) -> float
516  {
517  return static_cast<const SequenceTrackSegmentCurve<float> *>(&segment)->getValue(
518  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f);
519  }},
520  {RTTI_OF(glm::vec2), [](const SequenceTrackSegment &segment, int curveIndex, sequencecurveenums::ESegmentValueTypes segmentType) -> float
521  {
522  return static_cast<const SequenceTrackSegmentCurve<glm::vec2> *>(&segment)->getValue(
523  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f)[curveIndex];
524  }},
525  {RTTI_OF(glm::vec3), [](const SequenceTrackSegment &segment, int curveIndex, sequencecurveenums::ESegmentValueTypes segmentType) -> float
526  {
527  return static_cast<const SequenceTrackSegmentCurve<glm::vec3> *>(&segment)->getValue(
528  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f)[curveIndex];
529  }},
530  {RTTI_OF(glm::vec4), [](const SequenceTrackSegment &segment, int curveIndex, sequencecurveenums::ESegmentValueTypes segmentType) -> float
531  {
532  return static_cast<const SequenceTrackSegmentCurve<glm::vec4> *>(&segment)->getValue(
533  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f)[curveIndex];
534  }}
535  };
536  const float track_height = track.mTrackHeight * mState.mScale;
537 
538  assert(segmentBase.get_type().template is_derived_from<SequenceTrackSegmentCurve<T>>());
539  const auto &segment = static_cast<const SequenceTrackSegmentCurve<T> &>(segmentBase);
540 
541  for(int v = 0; v < segment.mCurves.size(); v++)
542  {
543  // calculate point of this value in the window
544  ImVec2 segment_value_pos =
545  {
546  trackTopLeft.x + segmentX - (segmentType == sequencecurveenums::BEGIN ? segmentWidth : 0.0f),
547  trackTopLeft.y + track_height * (1.0f - ((segmentType == sequencecurveenums::BEGIN ?
548  (float) segment.mCurves[v]->mPoints[0].mPos.mValue :
549  (float) segment.mCurves[v]->mPoints[
550  segment.mCurves[v]->mPoints.size() - 1].mPos.mValue) /
551  1.0f))
552  };
553 
554  bool hovered = false;
555 
556  if(mState.mIsWindowFocused)
557  {
558  // check if we are hovering this value
559  if((mState.mAction->isAction<sequenceguiactions::None>() ||
560  mState.mAction->isAction<sequenceguiactions::HoveringSegmentValue>() ||
561  mState.mAction->isAction<sequenceguiactions::HoveringSegment>() ||
562  mState.mAction->isAction<sequenceguiactions::HoveringCurve>()) &&
563  ImGui::IsMouseHoveringRect(
564  {segment_value_pos.x - (12.0f * mState.mScale),
565  segment_value_pos.y - (12.0f * mState.mScale)}, // top left
566  {segment_value_pos.x + (12.0f * mState.mScale),
567  segment_value_pos.y + (12.0f * mState.mScale)})) // bottom right
568  {
569  hovered = true;
570  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::HoveringSegmentValue>(
571  track.mID,
572  segment.mID,
573  segmentType,
574  v);
575 
576  if(ImGui::IsMouseDown(0))
577  {
578  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::DraggingSegmentValue>(
579  track.mID,
580  segment.mID,
581  segmentType,
582  v,
583  get_value_map[RTTI_OF(T)](segment, v, segmentType));
584  } else if(ImGui::IsMouseDown(1))
585  {
586  const auto &curve_segment = static_cast<const SequenceTrackSegmentCurve<T> &>(segment);
587  const auto &curve_track = static_cast<const SequenceTrackCurve<T> &>(track);
588 
589  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::OpenEditSegmentCurveValuePopup<T>>(
590  track.mID,
591  segment.mID,
592  segmentType,
593  v,
594  (segmentType == sequencecurveenums::ESegmentValueTypes::BEGIN)
595  ? curve_segment.getStartValue() : curve_segment.getEndValue(),
596  curve_track.mMinimum,
597  curve_track.mMaximum);
598  }
599 
600  showValue<T>(
601  track,
602  segment,
603  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f,
604  segmentType == sequencecurveenums::BEGIN ? segment.mStartTime : segment.mStartTime +
605  segment.mDuration,
606  v);
607  } else if(!mState.mAction->isAction<sequenceguiactions::DraggingSegmentValue>())
608  {
609  if(mState.mAction->isAction<sequenceguiactions::HoveringSegmentValue>())
610  {
611  auto *action = mState.mAction->getDerived<sequenceguiactions::HoveringSegmentValue>();
612 
613  if(action->mType == segmentType &&
614  action->mSegmentID == segment.mID &&
615  action->mCurveIndex == v)
616  {
617  mState.mAction = sequenceguiactions::createAction<sequenceguiactions::None>();
618 
619  showValue<T>(
620  track,
621  segment,
622  segmentType == sequencecurveenums::BEGIN ? 0.0f : 1.0f,
623  segmentType == sequencecurveenums::BEGIN ? segment.mStartTime : segment.mStartTime +
624  segment.mDuration,
625  v);
626  }
627  }
628  }
629 
630  // handle dragging segment value
631  if(mState.mAction->isAction<sequenceguiactions::DraggingSegmentValue>())
632  {
633  auto *action = mState.mAction->getDerived<sequenceguiactions::DraggingSegmentValue>();
634  if(action->mSegmentID == segment.mID)
635  {
636  if(action->mType == segmentType && action->mCurveIndex == v)
637  {
638  hovered = true;
639  showValue<T>(
