231  template<typename T>
232  std::string joinString(const T& list, const char* delim)
233  {
234  std::stringstream ss;
235  for (size_t i = 0, len = list.size(); i < len; i++)
236  {
237  if (i > 0)
238  ss << delim;
239  ss << list.at(i);
240  }
241  return ss.str();
242  }

245  template <typename... Args>
246  std::string stringFormat(const char* format, Args&&... args)
247  {
248  size_t size = (size_t)(snprintf(nullptr, 0, format, args...) + 1); // Extra space for '\0'
249  std::unique_ptr<char[]> buf(new char[size]);
250  snprintf(buf.get(), size, format, args...);
251  return std::string(buf.get(), buf.get() + size - 1); // We don't want the '\0' inside
252  }
