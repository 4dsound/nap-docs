149  std::vector<T*> Clipboard::deserialize(std::vector<std::unique_ptr<rtti::Object>>& createdObjects, utility::ErrorState& errorState)
150  {
151  std::vector<T*> deserialized_objects;
152 
153  createdObjects.clear();
154  for(auto& pair : mSerializedObjects )
155  {
156  // get the serialized object from the map
157  std::string serialized_object = pair.second;
158 
159  // De-serialize
160  rtti::DeserializeResult result;
161  rtti::Factory factory;
162  if (!rtti::deserializeJSON(serialized_object, rtti::EPropertyValidationMode::DisallowMissingProperties,
163  rtti::EPointerPropertyMode::NoRawPointers, factory, result, errorState))
164  return std::vector<T*>();
165 
166  // Resolve links
167  if (!rtti::DefaultLinkResolver::sResolveLinks(result.mReadObjects, result.mUnresolvedPointers, errorState))
168  return std::vector<T*>();
169 
170  // Make sure we deserialized something
171  T* root_object = nullptr;
172  if(!errorState.check(!result.mReadObjects.empty(), "No objects deserialized"))
173  return std::vector<T*>();
174 
175  auto* first_object = result.mReadObjects[0].get();
176  if( !errorState.check(first_object->get_type() == RTTI_OF(T) || first_object->get_type().is_derived_from<T>(), "Root object not of correct type"))
177  return std::vector<T*>();
178 
179  // Move ownership of read objects
180  root_object = static_cast<T*>(first_object);
181  for (auto& read_object : result.mReadObjects)
182  {
183  if (read_object->get_type().is_derived_from<T>())
184  root_object = dynamic_cast<T*>(read_object.get());
185  createdObjects.emplace_back(std::move(read_object));
186  }
187 
188  // init objects
189  for (auto& object_ptr : createdObjects)
190  {
191  if (!errorState.check(object_ptr->init(errorState), "Error initializing object : %s " , errorState.toString().c_str()))
192  return std::vector<T*>();
193  }
194 
195  if( !errorState.check(root_object != nullptr, "return object is null"))
196  return std::vector<T*>();
197 
198  deserialized_objects.emplace_back(root_object);
199  }
200 
201  return deserialized_objects;
202  }
203  }
