180  std::vector<T*> Clipboard::deserialize(std::vector<std::unique_ptr<rtti::Object>>& createdObjects, utility::ErrorState& errorState)
181  {
182  std::vector<T *> deserialized_objects;
183 
184  createdObjects.clear();
185  for(auto &pair: mSerializedObjects)
186  {
187  // get the serialized object from the map
188  std::string serialized_object = pair.second;
189 
190  // De-serialize
191  rtti::DeserializeResult result;
192  rtti::Factory factory;
193  if(!rtti::deserializeJSON(serialized_object, rtti::EPropertyValidationMode::DisallowMissingProperties,
194  rtti::EPointerPropertyMode::NoRawPointers, factory, result, errorState))
195  return std::vector<T *>();
196 
197  // Resolve links
198  if(!rtti::DefaultLinkResolver::sResolveLinks(result.mReadObjects, result.mUnresolvedPointers, errorState))
199  return std::vector<T *>();
200 
201  // Make sure we deserialized something
202  T *root_object = nullptr;
203  if(!errorState.check(!result.mReadObjects.empty(), "No objects deserialized"))
204  return std::vector<T *>();
205 
206  auto *first_object = result.mReadObjects[0].get();
207  if(!errorState.check(first_object->get_type() == RTTI_OF(T) ||
208  first_object->get_type().is_derived_from<T>(), "Root object not of correct type"))
209  return std::vector<T *>();
210 
211  // Move ownership of read objects
212  root_object = static_cast<T *>(first_object);
213  for(auto &read_object: result.mReadObjects)
214  {
215  createdObjects.emplace_back(std::move(read_object));
216  }
217 
218  // init objects
219  for(auto &object_ptr: createdObjects)
220  {
221  if(!errorState.check(object_ptr->init(errorState), "Error initializing object : %s ", errorState.toString().c_str()))
222  return std::vector<T *>();
223  }
224 
225  if(!errorState.check(root_object != nullptr, "return object is null"))
226  return std::vector<T *>();
227 
228  deserialized_objects.emplace_back(root_object);
229  }
230 
231  return deserialized_objects;
232  }
233  }
